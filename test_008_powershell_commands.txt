# RS232_WEB_CLOUD_008_DevicePullAck_SIM - PowerShell test sequence
# 1) In Render Environment set:
#    ALLOW_REMOTE_COMMANDS=1
#    COMMAND_QUEUE_ENABLED=1
#    COMMAND_TOKEN=your-command-token
#    DEVICE_TOKEN=your-device-token
# 2) Deploy latest commit.

$base = "https://rs232-web-cloud-001-render.onrender.com"
$deviceToken = "ТВОЯ_DEVICE_TOKEN"
$commandToken = "ТВОЯ_COMMAND_TOKEN"

# Check health
Invoke-RestMethod -Uri "$base/health"

# Queue set_next_no command
$cmdBody = Get-Content .\sample_command_request_set_next_no_008.json -Raw
Invoke-RestMethod `
  -Method Post `
  -Uri "$base/api/request-command" `
  -Headers @{ Authorization = "Bearer $commandToken" } `
  -ContentType "application/json" `
  -Body $cmdBody

# Simulate ESP32 pulling the command
$pull = Invoke-RestMethod `
  -Uri "$base/api/pull" `
  -Headers @{ Authorization = "Bearer $deviceToken" }
$pull

# Create ACK body with the real pulled command id
$ackObj = Get-Content .\sample_device_ack_set_next_no_ok.json -Raw | ConvertFrom-Json
$ackObj.id = $pull.command.id
$ackBody = $ackObj | ConvertTo-Json -Depth 20

# Simulate ESP32 acknowledging and returning updated state
Invoke-RestMethod `
  -Method Post `
  -Uri "$base/api/ack" `
  -Headers @{ Authorization = "Bearer $deviceToken" } `
  -ContentType "application/json" `
  -Body $ackBody

# Check latest state and queue/ack history
Invoke-RestMethod -Uri "$base/api/latest"
Invoke-RestMethod -Uri "$base/api/command-queue?ack=1"
Invoke-RestMethod -Uri "$base/api/ack-history"
